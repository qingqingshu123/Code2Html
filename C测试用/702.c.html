  580. void Game::Playing()
  581. {
  582. 	//HANDLE MFUN;
  583. 	//MFUN= CreateThread(NULL, 0, MusicFun, NULL, 0, NULL); 
  585. 	drawEnemy();
  586. 	drawPlane();
  588. 	int flag_bullet = 0;
  589. 	int flag_enemy = 0;
  591. 	while(1)
  592. 	{
  593. 		Sleep(8);
  594. 		if(_kbhit())
  595. 		{
  596. 			char x = _getch();
  597. 			if ('a' == x || 's' == x || 'd' == x || 'w' == x)
  598. 			{
  599. 				drawPlaneToNull();
  600. 				planeMove(x);
  601. 				drawPlane();
  602. 				judgePlane();
  603. 			}			
  604. 			else if ('p' == x)
  605. 				Pause();
  606. 			else if( 'k' == x)
  607. 				Shoot();
  608. 			else if( 'e' == x)
  609. 			{
  610. 				//CloseHandle(MFUN);
  611. 				GameOver();
  612. 				break;
  613. 			}
  614. 				
  615. 		}
  616. 		/* 处理子弹 
  617. /* 处理子弹 */
  619. 		if( 0 == flag_bullet )
  620. 		{
  621. 			bulletMove();
  622. 			drawBulletToNull();
  623. 			drawBullet();
  624. 			judgeEnemy();
  625. 		}			
  626. 		flag_bullet++;
  627. 		if( 5 == flag_bullet )
  628. 			flag_bullet = 0;
  630. 		/* 处理敌人 
  631. /* 处理敌人 */
  633. 		if( 0 == flag_enemy )
  634. 		{
  635. 			drawEnemyToNull();
  636. 			enemyMove();			
  637. 			drawEnemy();
  638. 			judgePlane();
  639. 		}
  640. 		flag_enemy++;
  641. 		if( flag_enemy >= rank )
  642. 			flag_enemy = 0;
  644. 		/* 输出得分 
  645. /* 输出得分 */
  647. 		printScore();
  648. 	}
  649. }
